Array.prototype.reduce 使用

Array.prototype.reduce 使用

語法

array.reduce(callback[(accumulator, currentValue, currentIndex, array)], initialValue)

使用

物件陣列轉 mapping 表

array.reduce((map, item) => Object.assign(map, { [item.id]: item.name }), {})

二維陣列轉一維

array.reduce((acc, cur) => {
    return acc.concat(cur)
}, [])

統計陣列中元素出現次數

array.reduce((acc, cur) => {
    if (cur in acc) {
        acc[cur] = 1
    } else {
        acc[cur] += 1
    }
    return acc
}, {})

物件陣列依條件分類

array.reduce((acc, cur) => {
    if (!acc[cur.type]) {
        acc[cur.type] = []
    }
    acc[cur.type].push(cur)
    return acc
}, {})

陣列去重

array.reduce((acc, cur) => {
    if (!acc.includes(cur)) {
        acc.push(cur)
    }
    return acc
}, [])

求最大值或最小值

array.reduce((acc, cur) => {
    if (!acc) {
        acc = cur
        return acc
    }
    if (acc.age < cur.age) {
        acc = cur
        return acc
    }
    return acc
}, 0)
